## Install

```bash
npm install --save crazybuttongroup
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'crazybuttongroup'
import 'crazybuttongroup/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
